    <script>

        function getKey() {

            return document.getElementById('gibier').value;

        }


        function charger() {

            const key = getKey();

            const data = JSON.parse(localStorage.getItem('huntstat_' + key)) || { tirs: 0, tues: 0 };

            document.getElementById('tirs').textContent = data.tirs;

            document.getElementById('tues').textContent = data.tues;

            calculerPrecision();

        }


        function modifier(type, valeur) {

            const id = type;

            let current = parseInt(document.getElementById(id).textContent);

            current = Math.max(0, current + valeur);

            document.getElementById(id).textContent = current;

            calculerPrecision();

        }


        function enregistrer() {

            const key = getKey();

            const tirs = parseInt(document.getElementById('tirs').textContent);

            const tues = parseInt(document.getElementById('tues').textContent);

            localStorage.setItem('huntstat_' + key, JSON.stringify({ tirs: tirs, tues: tues }));

            alert("Données enregistrées pour " + key);

        }


        function calculerPrecision() {

            const tirs = parseInt(document.getElementById('tirs').textContent);

            const tues = parseInt(document.getElementById('tues').textContent);

            const precision = tirs === 0 ? 0 : Math.round((tues / tirs) * 100);

            document.getElementById('precision').textContent = precision + '%';

        }


        document.getElementById('gibier').addEventListener('change', charger);

        window.onload = charger;
